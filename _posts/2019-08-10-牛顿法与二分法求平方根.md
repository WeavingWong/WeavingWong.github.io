
# 牛顿法求平方根


```python
# __*__ coding:utf8 __*__
import math
from math import sqrt
# 利用牛顿法求平方根
def sqrt_newton(a):
    if a < 1e-6:
        return 0
    x , y = (a,0)
    count = 0
    while abs(y-x) > 0.00000001:
        count += 1
        y = x
        print(count,'\t',y)
        x = 0.5*(x + a / x)
    return x
sqrt_newton(5)
```

# 二分法求平方根


```python
import math
def sqrt_by_dichotomy(n):
    if n < 0:
        return -1.0
    else:
        low = 0
        up = (n if n>=1 else 1) # 为防止初值n为小于1的数
        mid = (low + up) / 2;
        count=0
        while True:
            if mid**2>n:
                up = mid
            else:
                low = mid
            mid = (up+low)/2
            count +=1
            print(count,'\t',mid**2,'\t',mid)
            if (abs(mid**2 - n) < 0.0000001):
                break
        return mid
sqrt_by_dichotomy(5)
```
